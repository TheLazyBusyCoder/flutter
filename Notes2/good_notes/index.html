String

void main() {
	var first_name = "String";
	var two = first_name + " tow";
	var three = "${two * 5} $two";
	print(three.length);
	print(three.isEmpty);
	print(three.toUpperCase());
	print(three.trim());	// remove spaces. 
	print(three.substring(2,5));
	print("leo/harry".split('/'));	// ['leo' , 'harry'];
	// see dart doc to see all the methods. 
}

Numbers

void main() {
	int num1 = 5;
	double num2 = 4.5;
	var num3 = num1 + num2;
	print(num3);	// 9.5

	// parsing
	var num4 = num.parse('12');
	print(num4);

	// double
	var x = 0.5;
	print(x.isFinite);	// true
	print(x.isInfinite); // false
	print(x.isNan); // false;
	print(x.isNegative); // false;
	print(x.round());
	
	var y = x.toInt();
	print(y.toString());
	print(y.abs());

}
