Operator

// Arithmetic Operators
int a = 10;
int b = 5;
print('a + b = ${a + b}');  // 15
print('a - b = ${a - b}');  // 5
print('a * b = ${a * b}');  // 50
print('a / b = ${a / b}');  // 2.0
print('a ~/ b = ${a ~/ b}'); // 2
print('a % b = ${a % b}');  // 0

// Equality and Relational Operators
print('a == b: ${a == b}'); // false
print('a != b: ${a != b}'); // true
print('a > b: ${a > b}');   // true
print('a < b: ${a < b}');   // false
print('a >= b: ${a >= b}'); // true
print('a <= b: ${a <= b}'); // false

// Type Test Operators
print('a is int: ${a is int}');     // true
print('a is! String: ${a is! String}'); // true

// Bitwise and Shift Operators
int c = 2; // 10 in binary
int d = 3; // 11 in binary
print('c & d: ${c & d}'); // 10 in binary, 2 in decimal
print('c | d: ${c | d}'); // 11 in binary, 3 in decimal
print('c ^ d: ${c ^ d}'); // 01 in binary, 1 in decimal
print('~c: ${~c}');       // ...11111101 in binary, -3 in decimal
print('c << 1: ${c << 1}'); // 100 in binary, 4 in decimal
print('c >> 1: ${c >> 1}'); // 1 in binary, 1 in decimal

// Assignment Operators
int e = 5;
e += 2;  // e = e + 2
print('e += 2: $e');  // 7

// Logical Operators
bool x = true;
bool y = false;
print('x && y: ${x && y}'); // false
print('x || y: ${x || y}'); // true
print('!x: ${!x}');         // false

// Conditional Operators
int? f;
f ??= 10;
print('f ??= 10: $f'); // 10

// Cascade Notation
var list = [1, 2, 3]
..add(4)
..remove(2);
print('List after cascade: $list'); // [1, 3, 4]
